#Purchase History

select a.username Username, i.item_name Item, i.price Price
from user_account a, user_purchase u, purchase p, item i
where a.user_id=u.user_id 
and u.purchase_id=p.purchase_id
and i.item_id=p.item_id;

#Shop

select s.name Shop, i.item_name Item, i.description Description, i.rarity Rarity, i.price Price
from item i, shop s, shop_item b
where b.shop_id=s.shop_id
and i.item_id=b.item_id;

#Categorized Items

select c.name Category, i.item_name Item
from item i, category c, item_category b
where i.item_id=b.item_id
and c.category_id=b.category_id;

#sql query
 $sql = 'SELECT c.name, i.item_name, i.description, i.rarity, i.price
    from item i, category c, item_category b
    where i.item_id=b.item_id
    and c.category_id=b.category_id
    ORDER BY item_id';
